Pridelava

materiala
je
pridobivanje
semenskega
materiala
kmetijskih
rastlin
z
razmnoževanjem
oziroma
z

,
ki

.
S

kmetijskih
rastlin
se

samo

,
ki
so

v
register
dobaviteljev
.
Vrste

rastlin
so
:
poljščine
:
žita
,
krmne
rastline
,
krmna
in
sladkorna
pesa
,
oljnice
in
predivnice
,
krompir
,
zelenjadnice
,
hmelj
,
trta
,
sadne
rastline
,
okrasne
rastline
.
Med

,

ali

(

,

,

,

,

,

,

in

)
,

za
:

.

sort
,

ali
EU

.

,

.

.
