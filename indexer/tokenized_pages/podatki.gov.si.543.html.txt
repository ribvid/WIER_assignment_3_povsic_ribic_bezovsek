V

Razvoj

.
Razlogov

.
V

,

.
Odprti

,

.
Odprti

tako

,

pa

odpiranju

,

novih

ter
novih
oblik

,
ki

in
ji

hitrejši

do
informacij
.
Če
želite

podatke
o

,

,

,

.

)

(

)

do

.
Dostopnost
zbirk
(

,
seznanitev
)
je
praviloma
v
javnem
sektorju
že

prek

javnih

in

.
V

,

(

)
kot

(

)
ne

značaja
in

informacij

.
Pomembno

(

)

(

)

pri

.
Praviloma

,

organizacije
oziroma
drugi
državljani
,

jih
v

objavijo
v

,

,

ipd
.

)
